inicio

    repetir
        escrever "Quantos números pretende inserir? (N > 0)"
        ler N
    até N > 0

    positivos <- 0
    soma_total <- 0

    soma_neg <- 0
    cont_neg <- 0

    para i <- 1 até N passo 1 faça
        escrever "Introduza um número:"
        ler num

        soma_total <- soma_total + num

        se num > 0 então
            positivos <- positivos + 1
        senão se num < 0 então
            soma_neg <- soma_neg + num
            cont_neg <- cont_neg + 1
        fimse
    fimpara

    percentagem <- (positivos / N) * 100
    media_total <- soma_total / N

    se cont_neg > 0 então
        media_neg <- soma_neg / cont_neg
    senão
        media_neg <- 0   // ou "não existem números negativos"
    fimse

    escrever "Percentagem de números positivos: ", percentagem, "%"
    escrever "Soma total dos números: ", soma_total
    escrever "Média total dos números: ", media_total
    escrever "Soma dos números negativos: ", soma_neg
    escrever "Média dos números negativos: ", media_neg

fim
